import numpy as np

def sage_layer(H, A, W_self, W_neigh):
    """Une couche GraphSAGE-mean : agrège les voisins puis combine au self."""
    deg = A.sum(axis=1, keepdims=True)
    neigh = (A @ H) / np.clip(deg, 1, None)   # moyenne des voisins
    out = H @ W_self + neigh @ W_neigh         # self + voisinage
    return np.maximum(out, 0.0)                # ReLU

Une couche d'agrégation moyenne des voisins, combinée à l'état du nœud — le cœur du routage d'intentions sémantiques sur le graphe.

</> Black-Scholes — prix call/put + delta
from math import log, sqrt, exp, erf

def norm_cdf(x):
    return 0.5 * (1.0 + erf(x / sqrt(2.0)))

def black_scholes(S, K, T, r, sigma):
    d1 = (log(S / K) + (r + 0.5 * sigma ** 2) * T) / (sigma * sqrt(T))
    d2 = d1 - sigma * sqrt(T)
    call = S * norm_cdf(d1) - K * exp(-r * T) * norm_cdf(d2)
    put = K * exp(-r * T) * norm_cdf(-d2) - S * norm_cdf(-d1)
    return {"call": call, "put": put, "delta": norm_cdf(d1)}

import numpy as np

def gbm_paths(S0, mu, sigma, T, steps, n):
    dt = T / steps
    Z = np.random.standard_normal((n, steps))
    incr = (mu - 0.5 * sigma ** 2) * dt + sigma * np.sqrt(dt) * Z
    return S0 * np.exp(np.cumsum(incr, axis=1))
